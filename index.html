// 1. Inicializar el mapa const map = L.map('map').setView([38.345, -0.49], 14); // 2. Capa de mapa (CartoDB para evitar bloqueos 403) L.tileLayer('https://{s}.basemaps.cartocdn.com/light_all/{z}/{x}/{y}{r}.png', { attribution: '© OpenStreetMap' }).addTo(map); // 3. Cargar el archivo paradas.json fetch('paradas.json') .then(response => response.json()) .then(paradas => { paradas.forEach(p => { // Creamos un marcador por cada parada del JSON const marker = L.marker([p.lat, p.lon]).addTo(map); // Al hacer clic, mostramos el nombre y consultamos tiempo real marker.on('click', () => { document.getElementById('info').innerHTML = ` ${p.n}
ID Parada: ${p.id}
Consultando tiempo real...
`; obtenerTiempos(p.id); }); }); }) .catch(error => console.error('Error al cargar paradas:', error)); // 4. Función de tiempo real (Recordatorio: requiere Proxy CORS en GitHub Pages) async function obtenerTiempos(stopId) { const proxy = "https://cors-anywhere.herokuapp.com/"; const urlApi = `https://api.vectalia.es/v1/alicante/stops/${stopId}/estimations`; try { const response = await fetch(proxy + urlApi); const data = await response.json(); let html = ""; if (data && data.length > 0) { data.forEach(bus => { html += `🚍 L${bus.linea}: ${bus.minutos} min
`; }); } else { html = "Sin buses próximos."; } document.getElementById('tiempos').innerHTML = html; } catch (e) { document.getElementById('tiempos').innerHTML = "Error de conexión."; } }
